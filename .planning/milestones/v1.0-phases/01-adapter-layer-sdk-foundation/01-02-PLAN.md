```typescript
export interface BackendConfig {
  model?: string;
  tools: AgentTool[];
  systemMessage?: string;
  cwd: string;
  sessionId?: string;
  streaming?: boolean;
}

export interface BackendSessionHandle {
  readonly sessionId: string;
  send(prompt: string, attachments?: unknown[]): Promise<string>;
  subscribe(listener: (event: AgentEvent) => void): () => void;
  destroy(): Promise<void>;
  abort(): Promise<void>;
}

export interface SessionBackend {
  readonly name: string;
  initialize(): Promise<void>;
  createSession(config: BackendConfig): Promise<BackendSessionHandle>;
  resumeSession(sessionId: string, config: BackendConfig): Promise<BackendSessionHandle>;
  shutdown(): Promise<void>;
}
```

   - `get sessionId(): string` — returns `this.sdkSession.sessionId`.
   - `async send(prompt: string): Promise<string>` — calls `const result = await this.sdkSession.sendAndWait({ prompt })`, returns `result?.data?.content ?? ""`.

     - Calls `this.sdkSession.on((event: CopilotSessionEvent) => { ... })` which returns an unsubscribe function.
     - Inside the callback: call `translateCopilotEvent(event)`, if result is not null, call `listener(result)`.
     - Returns the SDK unsubscribe function.
   - `async destroy(): Promise<void>` — calls `await this.sdkSession.destroy()`.
   - `async abort(): Promise<void>` — calls `await this.sdkSession.abort()`.

   - Constructor takes `CopilotClientManager` instance.
   - `readonly name = "copilot"`.
   - `async initialize(): Promise<void>` — calls `await this.clientManager.start()`.
   - `async createSession(config: BackendConfig): Promise<BackendSessionHandle>`:
     - Gets client via `this.clientManager.getClient()`.
     - Bridges tools: `const sdkTools = bridgeAllTools(config.tools, {})`.
     - Creates session: `const session = await client.createSession({ model: config.model, streaming: config.streaming ?? true, tools: sdkTools, onPermissionRequest: approveAll, sessionId: config.sessionId, cwd: config.cwd })`.

   - `async resumeSession(sessionId: string, config: BackendConfig): Promise<BackendSessionHandle>`:
     - Gets client, bridges tools, calls `client.resumeSession(sessionId, { tools: sdkTools, onPermissionRequest: approveAll })`.
     - Returns `new CopilotSessionHandle(session)`.
   - `async shutdown(): Promise<void>` — calls `await this.clientManager.stop()`.
