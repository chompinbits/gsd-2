describe('Plan-Check Equivalence (Pi vs Copilot)', () => {
  
  it('should produce equivalent validation verdicts', async () => {
    const config = { /* minimal config */ };
    
    // Generate plans on both backends
    const piPlan = await runPlanWorkflow(config, { backend: 'pi' });
    const copilotPlan = await runPlanWorkflow(config, { backend: 'copilot' });
    
    // Run plan-check on both
    const piCheck = await validatePlan(piPlan.plan);
    const copilotCheck = await validatePlan(copilotPlan.plan);
    
    // Assertions:
    // 1. Both pass or both fail on critical issues (D-05, D-06)
    if (piCheck.criticalFailures.length > 0) {
      // If Pi has critical failures, Copilot should also have similar issues
      expect(copilotCheck.criticalFailures.length).toBeGreaterThan(0);
    } else {
      // If Pi passes critical, Copilot should also pass
      expect(copilotCheck.criticalFailures.length).toBe(0);
    }
    
    // 2. Plan scores within 10% (allows for minor estimation differences)
    const scoreDiff = Math.abs(piCheck.score - copilotCheck.score);
    expect(scoreDiff).toBeLessThan(10);
    
    // 3. Both pass or both fail (same overall verdict)
    expect(piCheck.passed).toBe(copilotCheck.passed);
  });
  
  it('should identify same critical issues on both paths', async () => {
    const config = { /* minimal config */ };
