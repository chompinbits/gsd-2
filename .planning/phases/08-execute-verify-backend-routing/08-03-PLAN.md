function resolvePlanningBackendFromSettings(): 'pi' | 'copilot' {
  try {
    const planningSettings = SettingsManager.create(process.cwd(), agentDir)
    return planningSettings.getDefaultBackend() ?? 'pi'
  } catch {
    return 'pi'
  }
