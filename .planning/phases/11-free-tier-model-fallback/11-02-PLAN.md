    **2. Add private method `_applyDowngradeIfNeeded`:**

    Add a private method to `CopilotSessionBackend`:
    ```typescript
    /**
     * Pre-flight downgrade check (D-03). Evaluates budget state and
     * substitutes the requested model with a 0× model if under pressure.
     * Returns the (possibly modified) config and the suggestion, or null if no downgrade.
     */
    private _applyDowngradeIfNeeded(config: BackendConfig): {
      config: BackendConfig;
      downgrade: DowngradeSuggestion | null;
    } {
      if (!this._accountingConfig || !this._currentTracker) {
        return { config, downgrade: null };
      }
      const budgetState = this._currentTracker.getState();
      const suggestion = suggestDowngrade(budgetState, this._accountingConfig);
      if (!suggestion) {
        return { config, downgrade: null };
      }
      // Override model in config (D-07: new sessions only, not mid-send)
      const downgradedConfig: BackendConfig = {
        ...config,
        model: suggestion.modelId,
      };
      return { config: downgradedConfig, downgrade: suggestion };
    }
    ```

    **3. Call in `createSession()`:**

    At the start of `createSession()`, after any existing setup but BEFORE calling the SDK to create the actual session, add:
    ```typescript
    const { config: effectiveConfig, downgrade } = this._applyDowngradeIfNeeded(config);
    if (downgrade) {
      // D-06: No silent fallback — emit structured notification
      console.error(
        `[gsd:accounting] ⚠ Model downgraded: ${config.model ?? "default"} → ${downgrade.modelId} (${downgrade.reason})`
      );
    }
    ```
    Then use `effectiveConfig` instead of `config` for the rest of `createSession()`.

    **4. Call in `resumeSession()`:**

    Same pattern as createSession. After setup, before session creation:
    ```typescript
    const { config: effectiveConfig, downgrade } = this._applyDowngradeIfNeeded(config);

      console.error(
        `[gsd:accounting] ⚠ Model downgraded: ${config.model ?? "default"} → ${downgrade.modelId} (${downgrade.reason})`
      );
    }
    ```
    Use `effectiveConfig` for remaining session logic.

    **Important:** Do NOT add downgrade logic inside `AccountingSessionHandle.send()` — that would violate D-07/D-08 (new sessions only, no mid-send swap). The check is only at session-creation time.
